@Controller
@Slf4j
@RequestMapping("/arith")
public class ArithmeticController {
   private CalculateService calculateService;
   @Autowired
   public ArithmeticController(CalculateService calculateService) {
       this.calculateService = calculateService;
   }
   @GetMapping("/")
   public String index() {
       return "arithmetic";
   }
   @PostMapping("/calcu")
   public String calcu(@ModelAttribute Equation equation) {
       log.debug("算式: {} {} {}",equation.getOp1(),equation.getOpr(),equation.getOp2());
